


RODATA က Read-Only Data ရဲ့အတိုကောက်ဖြစ်တယ်
Read-only section ဖြစ်တယ် → runtime မှာ ပြောင်းလို့မရဘူး
global Constant data တွေ သိမ်းဆည်းတယ်
Text segment နဲ့ အတူ (သို့) သီးခြားစီ ရှိတတ်တယ်
Program run နေစဉ် modify လုပ်ရင် segmentation fault တက်တယ်

```
High address
+-----------------+
|      Stack      |  (local variables, return addresses)
|        ↓        |
|        ↑        |
|      Heap       |  (malloc)
+-----------------+
|      .bss       |  (uninitialized, RW)
+-----------------+
|      .data      |  (initialized, RW)
+-----------------+
|     .rodata     |  ← read-only data (strings, constants)
+-----------------+
|      .text      |  ← code (read-only, executable)
+-----------------+
Low address
```
`.rodata` က `.text` နဲ့ `.data` ကြားမှာ (သို့) `.text` ရဲ့အပေါ်မှာ ရှိတယ်။ **Permission** က **read-only** (no write, no execute ပုံမှန်အားဖြင့်


1. **String Literals

    ```c
    char *str = "Hello World"; // "Hello World" ဆိုတဲ့ စာသားကြီးက .rodata ထဲ ရောက်ပါတယ်။
    ```
    
2. **Global Const Variables:**

    ```c
    const int global_max = 100; // Global အဆင့်မို့လို့ .rodata ထဲ ရောက်ပါတယ်။
    ```
    
3. **Static Const Variables (Local ကော Global ကော):**
    
    ```c
    static const float PI = 3.14159; // .rodata ထဲ ရောက်ပါတယ်။
    ```
    
4. **Lookup Tables / Constant Arrays:** ပရိုဂရမ်တစ်ခုလုံးမှာ အပြောင်းအလဲမရှိ သုံးမယ့် data table တွေကို `static const` ပေးထားရင် `.rodata` ထဲ ရောက်သွားမှာ ဖြစ်ပါတယ်။
    
    ```c
    static const int hex_lookup[] = {0, 1, 2, 3, 4, 5, 6, 7}; 
    ```
    

```c
// ============================================
// ALL DATA STORED IN RODATA SEGMENT
// ============================================

// 1. GLOBAL CONST (any global const)
const int GLOBAL_CONST = 100;           // ✅ .rodata
const float GLOBAL_CONST_FLOAT = 3.14;  // ✅ .rodata
const char GLOBAL_CONST_CHAR = 'A';     // ✅ .rodata (single char also in .rodata)
const double GLOBAL_CONST_DOUBLE = 2.71; // ✅ .rodata

// 2. STATIC CONST (global or local scope, but file-scope here)
static const int STATIC_CONST = 200;     // ✅ .rodata (file scope)
static const char STATIC_CONST_STR[] = "static const string"; // ✅ .rodata (entire array)

// 3. STRING LITERALS (all quoted strings)
char *p1 = "hello world";               // "hello world" → .rodata, p1 pointer → Data
char *p2 = "abc";                       // "abc" → .rodata, p2 pointer → Data
const char *p3 = "readonly";            // "readonly" → .rodata, p3 pointer → Data
char *arr[] = {"one", "two", "three"};  // strings → .rodata, arr array (pointers) → Data

// 4. ENUM constants (compile-time, no memory allocated)
enum Colors { RED = 1, GREEN = 2, BLUE = 3 };  
// ⚠️ Enums don't have memory address! They're compile-time constants.
// Values are inlined directly into code (.text), not stored in .rodata.

// 5. CONST on global array
const int GLOBAL_CONST_ARR[3] = {10, 20, 30};  // ✅ .rodata (entire array)

// 6. CONST struct (global)
const struct Point {
    int x;
    int y;
} GLOBAL_CONST_POINT = {5, 10};                // ✅ .rodata (entire struct)

void function() {
    // 7. LOCAL STATIC CONST
    static const int local_static_const = 500;  // ✅ .rodata (not stack!)
    // Static variables always go to Data/BSS/.rodata, never stack.
    
    // 8. LOCAL CONST (actually goes to stack, not rodata!)
    const int local_const = 50;                 // ⚠️ STACK, NOT .rodata!
    // ⚠️⚠️ IMPORTANT: Local const is on STACK, but compiler enforces read-only
    // at compile-time. At runtime, you CAN modify it (undefined behavior, but no crash).
    
    // 9. STRING LITERALS inside function
    char *local_ptr = "local literal";          // "local literal" → .rodata  (pointers) → Stack*
    // local_ptr pointer itself is on STACK, pointing to .rodata 
    
    // 9b. Local const array (stack copy!)
    const char local_arr[] = "hello";           // ⚠️ STACK (copy from .rodata)
    // Even though it's const, the array is stored on stack as a copy!
    
    // 9c. Local static const string (rodata)
    static const char *static_str = "static literal";  // pointer → Data, literal → .rodata
}
```

#### Not in `rodata`

|Data Type|Storage|Why not rodata?|
|---|---|---|
|`const int x = 5;` (local)|**STACK**|Local const က compiler က enforce လုပ်ပေမယ့် stack မှာပဲထား|
|`char arr[] = "hello";`|STACK or DATA|Array က copy ယူတယ်, literal က rodata မှာရှိပေမယ့် arr က copy|
|`int x = 5;` (non-const)|STACK or DATA|Mutable variable|
|`static int x = 5;` (non-const)|DATA/BSS|Mutable static|
|`malloc()`|HEAP|Dynamic memory|

```c
const int GLOBAL_CONST = 100;   // ✅ rodata (read-only, segfault if modified)

int main() {
    const int LOCAL_CONST = 100;  // ⚠️ STACK (not rodata!)
    
    // ဒီနှစ်ခု လုံးဝမတူဘူး!
    
    // GLOBAL_CONST = 200;  // ❌ Compile error (or segfault if bypassed)
    // LOCAL_CONST = 200;   // ❌ Compile error (but can be overwritten by overflow!)
    
    char buf[4];
    gets(buf);  // overflow လုပ်ရင် LOCAL_CONST ကို ပြောင်းနိုင် (stack မှာရှိတယ်)
}
```

local const က stack မှာ → overflow နဲ့ ပြောင်းလို့ရတယ် 
Global const က `rodata` မှာ → မပြောင်းရဘူး (`segfault`)


```c
#include <stdio.h>

// ALL of these are in rodata
const int c1 = 10;
const int c2 = 20;
const char c3 = 'X';
const char *str1 = "constant string";
static const int sc1 = 30;
static const char sc2[] = "static constant array";

int main() {
    // NOT in rodata (stack)
    const int local_const = 40;
    
    // Pointer to rodata (pointer on stack, points to rodata)
    const char *local_ptr = "local literal";
    
    printf("rodata addresses:\n");
    printf("&c1: %p\n", &c1);
    printf("&c2: %p\n", &c2);
    printf("&c3: %p\n", &c3);
    printf("str1 (points to): %p\n", str1);
    printf("&sc1: %p\n", &sc1);
    printf("sc2 (array): %p\n", sc2);
    printf("\"local literal\": %p\n", "local literal");
    printf("\"constant string\": %p\n", "constant string");
    
    printf("\nNOT in rodata:\n");
    printf("&local_const (stack): %p\n", &local_const);
    printf("&local_ptr (stack): %p\n", &local_ptr);
    
    return 0;
}
```

```c
Compile-time (Program မ run ခင်):
.rodata မှာ:
┌─────────────────────────────┐
│  'h' 'e' 'l' 'l' 'o' '\0'   │  ← Original string literal
│  address: 0x400600           │
└─────────────────────────────┘

Run-time (function စခေါ်တဲ့အခါ):
1. Stack ပေါ်မှာ local_arr အတွက် နေရာ 6 bytes (including '\0') ခွဲတယ်
2. .rodata ကနေ data တွေကို stack ပေါ်ကို copy လုပ်တယ်

Stack ပေါ်မှာ:
┌─────────────────────────────┐
│  'h' 'e' 'l' 'l' 'o' '\0'   │  ← Copy! (different address)
│  address: 0x7ffd1234         │
└─────────────────────────────┘

ရလဒ်: Same content, but DIFFERENT memory locations!
```

```c
#include <stdio.h>
#include <string.h>

int main() {
    // String literal (in .rodata)
    const char *ptr = "hello";
    
    // Local array (copy from .rodata to stack)
    const char local_arr[] = "hello";
    
    printf("=== ADDRESSES ===\n");
    printf("'hello' literal address: %p\n", "hello");
    printf("ptr points to: %p\n", ptr);
    printf("local_arr address: %p\n", local_arr);
    
    printf("\n=== ARE THEY THE SAME? ===\n");
    if (ptr == local_arr) {
        printf("Same address!\n");
    } else {
        printf("DIFFERENT addresses!\n");
        printf("  .rodata: %p\n", ptr);
        printf("  Stack:   %p\n", local_arr);
    }
    
    printf("\n=== CONTENT (same) ===\n");
    printf("ptr: %s\n", ptr);
    printf("local_arr: %s\n", local_arr);
    
    printf("\n=== MODIFICATION TEST ===\n");
    // Modify the stack copy (works)
    // local_arr[0] = 'H';  // ⚠️ Warning but works (on stack)
    // printf("Modified local_arr: %s\n", local_arr);  // "Hello"
    
    // Modify the literal (CRASH!)
    // ptr[0] = 'H';  // ❌ Segmentation fault!
    
    return 0;
}
```
